Příloha C: Alternativní XML Schéma: W3C XML Schema 1.0 (XSD)¶

W3C XML Schema 1.0 je dnes pravděpodobně nejpopulárnější formát pro popis XML struktur.

Níže uvedené schéma bylo vytvořeno transformací z RNC schématu a mělo by zobrazit všechna pravidla z RNC. V (nepravděpodobném) případě, že se vyskytnou rozdíly, považujte RELAX NG RNC za referenční verzi.

schema-cs - kořen schématu¶

<?xml version="1.0" encoding="UTF-8"?>
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema" elementFormDefault="qualified" targetNamespace="x-format:cz-ndic_ddr-fcd-v2.0" xmlns:c="x-format:cz-ndic_ddr-fcd-v2.0">
  <xs:element name="DOC">
    <xs:annotation>
      <xs:documentation>kořenový element publikace</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:sequence>
        <xs:element ref="c:INF"/>
        <xs:element ref="c:FCD"/>
      </xs:sequence>
      <xs:attribute name="created" use="required" type="xs:dateTime">
        <xs:annotation>
          <xs:documentation>čas vytvoření dokumentu</xs:documentation>
        </xs:annotation>
      </xs:attribute>
    </xs:complexType>
  </xs:element>
  <xs:element name="INF">
    <xs:annotation>
      <xs:documentation>kontextové informace k publikaci</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:sequence>
        <xs:element ref="c:DAT"/>
      </xs:sequence>
    </xs:complexType>
  </xs:element>
  <xs:element name="DAT">
    <xs:annotation>
      <xs:documentation>informace o souvisejících datových sadách</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:sequence>
        <xs:element ref="c:D2PLS"/>
      </xs:sequence>
    </xs:complexType>
  </xs:element>
  <xs:element name="D2PLS">
    <xs:annotation>
      <xs:documentation>DATEX II sada předdefinovaných míst</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:attribute name="uri" use="required" type="xs:NMTOKEN">
        <xs:annotation>
          <xs:documentation>uri zdroje (implicitně nebo explicitně zahrnuje i verzi)</xs:documentation>
        </xs:annotation>
      </xs:attribute>
    </xs:complexType>
  </xs:element>
  <xs:element name="FCD">
    <xs:annotation>
      <xs:documentation>kontejner pro informace z jednotlivých úseků</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:group maxOccurs="unbounded" ref="c:SEGMENT"/>
    </xs:complexType>
  </xs:element>
  <xs:group name="SEGMENT">
    <xs:sequence>
      <xs:element maxOccurs="unbounded" ref="c:SEGMENT"/>
    </xs:sequence>
  </xs:group>
  <xs:element name="SEGMENT">
    <xs:annotation>
      <xs:documentation>informace pro jeden úsek</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:sequence>
        <xs:element ref="c:ALL"/>
        <xs:element minOccurs="0" ref="c:CAR"/>
        <xs:element minOccurs="0" ref="c:TRUCK"/>
      </xs:sequence>
      <xs:attribute name="id" use="required" type="xs:NCName">
        <xs:annotation>
          <xs:documentation>identifikátor úseku ze sady předdefinovaných míst</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="version" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>verze popisu polohy segmentu s daným id</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="last_updated" use="required" type="xs:dateTime">
        <xs:annotation>
          <xs:documentation>čas, pro který jsou data vypočtena</xs:documentation>
        </xs:annotation>
      </xs:attribute>
    </xs:complexType>
  </xs:element>
  <xs:element name="ALL">
    <xs:annotation>
      <xs:documentation>hodnoty bez rozlišení typů vozidel (sdružený dopravní proud)</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:sequence>
        <xs:element minOccurs="0" maxOccurs="1" ref="c:CONGESTION"/>
      </xs:sequence>
      <xs:attribute name="count" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>počet vozidel použitých pro výpočet hodnot</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="speed" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>rychlost dopravního proudu [km/h]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="travel_time" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>dojezdová doba [s]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="delay" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>zpoždění [s]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="traffic_level" use="required">
        <xs:annotation>
          <xs:documentation>stupeň dopravy dle zvyklostí v ČR

https://cs.wikipedia.org/wiki/Stupe%C5%88_provozu</xs:documentation>
        </xs:annotation>
        <xs:simpleType>
          <xs:restriction base="xs:token">
            <xs:enumeration value="1">
              <xs:annotation>
                <xs:documentation>stupeň 1: provoz pouze jednotlivých vozidel, jízda je plynulá</xs:documentation>
              </xs:annotation>
            </xs:enumeration>
            <xs:enumeration value="2">
              <xs:annotation>
                <xs:documentation>stupeň 2: provoz malých skupinek vozidel, jízda je plynulá, odbavování na křižovatkách bez problémů</xs:documentation>
              </xs:annotation>
            </xs:enumeration>
            <xs:enumeration value="3">
              <xs:annotation>
                <xs:documentation>stupeň 3: tvoří se proudy vozidel, provoz je plynulý, avšak rychlost nižší než maximální povolená</xs:documentation>
              </xs:annotation>
            </xs:enumeration>
            <xs:enumeration value="4">
              <xs:annotation>
                <xs:documentation>stupeň 4: tvoří se kolony vozidel, provoz není plynulý, průměrná rychlost je výrazně snížená, průjezd křižovatkami je narušen</xs:documentation>
              </xs:annotation>
            </xs:enumeration>
            <xs:enumeration value="5">
              <xs:annotation>
                <xs:documentation>stupeň 5: dopravní kolaps – vozidla na komunikacích stojí nebo v kolonách jen velmi pomalu popojíždějí, průměrná rychlost je velmi malá</xs:documentation>
              </xs:annotation>
            </xs:enumeration>
          </xs:restriction>
        </xs:simpleType>
      </xs:attribute>
      <xs:attribute name="freeflow_speed" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>rychlost volného průjezdu [km/h]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="freeflow_time" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>doba volného průjezdu [s]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="reliability" use="required" type="xs:decimal">
        <xs:annotation>
          <xs:documentation>spolehlivost dat, která charakterizuje kvalitu datového vzorku

Nabývá hodnot &lt;0-1&gt; kdy 1 znamená 100% spolehlivá data.
Kvalita datového vzorku se hodnotí podle stáří použitých vstupních poloh, jejich počtu atp. Vzorec výpočtu není součástí tohoto profilu.</xs:documentation>
        </xs:annotation>
      </xs:attribute>
    </xs:complexType>
  </xs:element>
  <xs:element name="CAR">
    <xs:annotation>
      <xs:documentation>hodnoty pro osobní vozidla</xs:documentation>
    </xs:annotation>
    <xs:complexType>
	  <xs:sequence>
        <xs:element minOccurs="0" maxOccurs="1" ref="c:CONGESTION"/>
      </xs:sequence>
      <xs:attribute name="count" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>počet vozidel použitých pro výpočet hodnot</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="speed" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>rychlost dopravního proudu [km/h]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="freeflow_speed" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>rychlost volného průjezdu [km/h]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
    </xs:complexType>
  </xs:element>
  <xs:element name="TRUCK">
    <xs:annotation>
      <xs:documentation>hodnoty pro nákladní vozidla</xs:documentation>
    </xs:annotation>
    <xs:complexType>
	  <xs:sequence>
        <xs:element minOccurs="0" maxOccurs="1" ref="c:CONGESTION"/>
      </xs:sequence>
      <xs:attribute name="count" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>počet vozidel použitých pro výpočet hodnot</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="speed" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>rychlost dopravního proudu [km/h]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="freeflow_speed" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>rychlost volného průjezdu [km/h]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
    </xs:complexType>
  </xs:element>
  <xs:element name="CONGESTION">
    <xs:annotation>
      <xs:documentation>Popis konkrétní kolony

Segmenty jsou interně děleny na menší části (popis jinde).
Na kterékoliv z těchto interních části vzniká kolona,
pokud na ní rychlost klesne pod 30 % rychlosti volného průjezdu.</xs:documentation>
    </xs:annotation>
    <xs:complexType>
      <xs:attribute name="from" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>začátek kolony [m] měřeno od počátku segmentu ve směru dopravního proudu</xs:documentation>
        </xs:annotation>
      </xs:attribute>
      <xs:attribute name="length" use="required" type="xs:integer">
        <xs:annotation>
          <xs:documentation>délka kolony [m]</xs:documentation>
        </xs:annotation>
      </xs:attribute>
    </xs:complexType>
  </xs:element>
</xs:schema>
