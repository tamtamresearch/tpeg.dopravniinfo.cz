Příloha A: Příklady zpráv¶

fcd-data-v2-01.xml - ukázka možných kombinací stavů¶

Zpráva obsahuje několik různých kombinací stavů na segmentech podle toho jestli jde o sdružený či dělený dopravní proud a podle toho jestli se na úseku vyskytuje kongesce.

<?xml version="1.0" encoding="UTF-8"?>
<DOC xmlns="x-format:cz-ndic_ddr-fcd-v2.0" created="2023-01-23T18:28:18+01:00">
  <INF>
    <DAT>
      <D2PLS uri="x-source:cz-ndic_d2-pls-fcd"/>
    </DAT>
  </INF>
  <FCD>
    <!-- sdružený dopravní proud bez kongesce -->
    <SEGMENT id="TS42739T17195" version="1" last_updated="2023-01-23T18:26:06+01:00">
      <ALL count="1" speed="41" travel_time="30" delay="5" freeflow_speed="50" freeflow_time="25" reliability="0.500" traffic_level="2"/>
    </SEGMENT>
    <!-- sdružený dopravní proud s kongescí -->
    <SEGMENT id="TS17576T42659" version="1" last_updated="2023-01-23T18:24:06+01:00">
      <ALL count="1" speed="5" travel_time="193" delay="175" freeflow_speed="50" freeflow_time="18" reliability="0.500" traffic_level="5">
        <CONGESTION from="98" length="171"/>
      </ALL>
    </SEGMENT>
    <!-- dělený dopravní proud s osobními vozidly a nákladními vozidly a s kongescí nákladních vozidel-->
    <SEGMENT id="TS22988T22989" version="1" last_updated="2023-01-23T18:28:05+01:00">
      <ALL count="3" speed="17" travel_time="484" delay="383" freeflow_speed="82" freeflow_time="101" reliability="0.750" traffic_level="4">
        <CONGESTION from="0" length="1936"/>
      </ALL>
      <CAR count="1" speed="6" freeflow_speed="81"/>
      <TRUCK count="2" speed="24" freeflow_speed="79">
        <CONGESTION from="0" length="1936"/>
      </TRUCK>
    </SEGMENT>
    <!-- dělený dopravní proud s osobními vozidly a nákladními vozidly a s kongescí nákladních i osobních vozidel-->
  <SEGMENT id="TS22989T22988" version="1" last_updated="2023-01-23T18:28:05+01:00">
    <ALL count="5" speed="37" travel_time="78" delay="52" freeflow_speed="114" freeflow_time="26" reliability="0.900" traffic_level="4">
      <CONGESTION from="220" length="232"/>
    </ALL>
    <CAR count="2" speed="35" freeflow_speed="114">
      <CONGESTION from="320" length="112"/>    
    </CAR>
    <TRUCK count="3" speed="43" freeflow_speed="80">
      <CONGESTION from="220" length="225"/>
    </TRUCK>
  </SEGMENT>
    <!-- dělený dopravní proud s osobními vozidly a nákladními vozidly -->
    <SEGMENT id="TS01225T25680" version="1" last_updated="2023-01-23T18:28:05+01:00">
      <ALL count="5" speed="112" travel_time="105" delay="14" freeflow_speed="130" freeflow_time="91" reliability="0.900" traffic_level="1"/>
      <CAR count="3" speed="128" freeflow_speed="130"/>

</DOC>
