Příloha B: Reprezentativní XML Schéma RELAX NG¶

K popisu struktury XML Je použito RELAX NG schéma v kompaktním formátu. Doporučujeme zkusit dané schéma přečíst, protože se nám tato forma jeví jako nejkompaktnější a nejčitelnější z dostupných typů schémat.

Pokud však trváte na použití W3C XML Schema 1.0, přikládáme i tento typ schématu, které je vytvořeno konverzí z RELAX NG.

RELAX NG Compact Syntax Schema (RNC)¶

RNC schéma popisuje očekávané struktury XML dokumentu.

Kořenový element je uveden pomocí start =.

Všechny další <SomeName> = elementy definují tzv. vzory.

V našem schématu jsou následující sekce:

  • prolog (deklarace jmenného prostoru)
  • vložení jiných schéma (není využito v tomto případě)
  • definice kořenového elementu (start =)
  • vzory pro elementy
  • vzory pro atributy
  • vzory pro datové typy

Komentáře ## poskytují více detailů k významu jednotlivých částí.

schem-cs.rnc - kořen schématu¶

default namespace = ""
## Varování: schéma bylo vytvořeno odvozením ze vzorků

## Kořenový prvek dokumentu
start =

element msg {
  attribute type { "data" },

  ## Čas publikace dokumentu
  element datetime { xsd:dateTime },

  ## Identifikátor odesílatele
  element sender { "JSDI_TDDR" },

  ## Identifikátor příjemce
  element receiver { "DACSServerADD" },

  ## Metoda přenosu
  element transmission { "HTTP SOAP" },

  ## Data naměřená jedním detektorem
  element detector {

    ## Identifikátor detektoru
    attribute id { xsd:integer },

    ## Identifikátor sady měření
    element data_id { type-guid },

    ## Identifikátor stavu měření
    element status_generalized { 
      
      ## OK
      "1" |

      ## Chyba
      "2" |

      ## Zařízení nezobrazuje požadovaný text nebo znak
      "3" |

      ## Mimo provoz
      "4" |

      ## Neznámý stav
      "5" |

      ## Ztráta komunikace se zařízením
      "6" |

      ## Ztráta komunikace s komunikačním modulem
      "7"
    },

    ## popis měření, jakýkoliv text
    element description { text },

    ## Čas výpočtu měřených hodnot
    element datetime { type-datetime-spaced },

    ## Čas začátku měřící periody
    element start { type-datetime-spaced },

    ## Čas konce měřící periody
    element stop { type-datetime-spaced },

    # Pokud je `status_generalized` 1, musí následující skupina být přítomna
    (

    ## Intenzita vozidel [-] vozidla za měřenou dobu
    element count { all, class+ },

    ## Obsazenost detektoru [%] 
    element occupancy { all, class+ },

    ## Rychlost vozidel [km/h]
    element speed { all, class+ }
    )?,

    ## Seznam chybových hlášení
    element error_list {

      ## Jednotlivá chybová hlášení
      ## Pozn: jednotliví dodavatelé dat užívají některé kódy chyb odlišně.
      element error {

        ## Kód chyby
        attribute id { xsd:integer },

        ## Popis chyby
        attribute description { text }
      }*
    }
  }+
}

all = element all { xsd:integer }

class =
  element class {

    ## Identifikátor kategorie vozidla
    attribute id {

      ## nezařazené vozidlo
      "0" |

      ## osobní auto 
      "1" |
      
      ## osobní auto s přívěsem 
      "2" |

      ## nákladní auto 
      "3" |

      ## nákladní auto s přívěsem 
      "4" |

      ## dodávka (Pickup) 
      "5" |

      ## autobus 
      "6" |

      ## motocykl 
      "7" |

      ## zvláštní vozidla 
      "8" |

      ## kamión (tahač s návěsem) 
      "9" |

      ## Osobní auto a podobná vozidla 
      "10" |

      ## Nákladní auto a podobná vozidla 
      "11" |

      ## Osobní (LN1 - kategorie1 0-5m) 
      "12" |

      ## Nákladní (LN2 - kategorie2 5-9m) 
      "13" |

      ## Autobusy (LN3 - kategorie3 9-12m) 
      "14" |

      ## Kamiony (LN4 - kategorie4 12m a vice) 
      "15" |

      ## Osobní automobil, motocykl, malá dodávka 
      "16" |

      ## Osobní automobil s přívěsem 
      "17" |

      ## Nákladní automobil 
      "18" |

      ## Nákladní automobil s přívěsem, kamion 
      "19" |

      ## Autobus 
      "20" |

      ## Ostatní, neidentifikované vozidlo 
      "21" |

      ## Celkovy pocet 
      "22" |

      ## Težké nákladní auto 
      "23" |

      ## Tahač s návěsem 
      "24" |

      ## tahač návěsů + nákladní přívesová souprava 
      "25" |

      ## motorky (LN1) 
      "26" |

      ## osobní vozidla (LN2) 
      "27" |

      ## dodávky a malá nákladní vozidla (LN3) 
      "28" |

      ## nákladní vozidla a autobusy (LN4 - 10,5m a vice)
      "29"  
     },

    ## Naměřená hodnota
    xsd:integer
  }

## GUID in format 8-4-4-4-12
type-guid =
  xsd:string { pattern = "[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}" }


## datetime string as xsd:string, but using space instead of "T"
type-datetime-spaced = 
  attribute format { "yyyy-MM-dd HH:mm:ssK" }?,
  xsd:string { pattern = "\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}[\+\-]\d{2}:\d{2}" }
