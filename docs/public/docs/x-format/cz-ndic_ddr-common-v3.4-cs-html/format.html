Příloha B: Reprezentativní XML Schéma RELAX NG¶

K popisu struktury XML Je použito RELAX NG schéma v kompaktním formátu. Zároveň je k dispozici i tradiční W3C XML Schema 1.0, vytvořené konverzí z RELAX NG.

RELAX NG Compact Syntax Schema (RNC)¶

RNC schéma popisuje očekávané struktury XML dokumentu.

Kořenový element je uveden pomocí start =.

Všechny další <SomeName> = elementy definují tzv. vzory.

V našem schématu jsou následující sekce:

  • prolog (deklarace jmenného prostoru)

  • vložení jiných schéma (není využito v tomto případě)

  • definice kořenového elementu (start =)

  • vzory pro elementy

  • vzory pro atributy

  • vzory pro datové typy

Komentáře ## poskytují více detailů k významu jednotlivých částí.

schema-cs.rnc - kořen schématu¶

default namespace = "x-format:cz-ndic_ddr-common-v3.4"

namespace openlr = "http://www.openlr.org/openlr"

start =

## Kořenový element dokumentu pro výměnu informací
element DOC {

    ## Jednoznačný identifikátor dokumentu
    attribute id { type-guid-8-4-4-4-12 },

    ## Kód země, ve které se události obsažené ve zprávě nachází
    ##
    ## V jedné zprávě se nesmí nacházet událostí z více států.
    attribute country { type-country },

    ## Subjekty, které si vyměnují XML dokument
    element INF {

        ## Metadata použitých datových sad
        element DAT {

            ## Metadata katalogu událostí ALERT-C
            element EVTT {

                ## Verze katalogu událostí ALERT-C
                attribute version { type-AlertCEventTableVersion }
            },

            ## Metadata lokační tabulky ALERT-C
            element LOCT {

                ## Verze lokační tabulky
                attribute version { type-AlertCLocationTableVersion },

                ## Číslo lokační tabulky
                attribute number { xsd:positiveInteger },

                ## Číslo země podle číselníku ALERT-C
                attribute country { xsd:positiveInteger }
            },

            ## Metadata datové sady sítě komunikací, na kterou jsou
            ## lokalizovány dopravní události
            element SNET {

                ## Typ použité datové sady sítě komunikací
                attribute type { type-snet-type },

                ## Verze datové sady sítě komunikací
                attribute version { type-networkVersion },

                ## Země, kterou datová sada sítě komunikací pokrývá
                attribute country { "CZ" }
            },

            ## Metadata datové sady UIRADR
            ## Dokumentace dovoluje uvézt pouze v případě, že zpráva obsahuje
            ## české události s vyplněnou informací o místu události v souladu
            ## s republikovým číselníkem UIRADR.
            element UIRADR {

                ## Verze struktury UIRADR
                attribute structure { text },

                ## Verze dat UIRADR
                attribute version { type-nonZeroDecimal },

                ## Datum vydání verze dat UIRADR
                attribute date { xsd:date }?
            }
        }
    },

    ## Kontejner pro všechny dílčí zprávy
    element MJD {

        element_MSG*
    }
}

## ======================================================

element_MSG =

## Zpráva
##
## Popis aktuálního stavu jedné události.
element MSG {

    ## Jednoznačný identifikátor zprávy
    attribute id { type-guid-max38 },

    ## Verze zprávy
    ##
    ## Nová = 1, někdy 0. Aktualizace, vždy +1
    attribute version { xsd:nonNegativeInteger },

    ## Typ poskytované informace
    ##
    ## Pozn.:  typ "WCOND" je popsán jako samostatný formát
    attribute type {

		## Traffic Information: Běžné dopravní informace
		"TI" |

		## Traffic Level: dopravní intenzita
		"TL"
	},

    ## Typ organizace poskytující dopravní informace
    ##
    ## Alfanumerický kód bez diakritiky.
    ## Příklad: "SSU"
    attribute provider { text },

    ## Zda byla událost ověřena na místě
    ##
    ## Většina zpráv je ověřena, ale např. u nezabezpečeného místa nehody bývá
    ## false až do ověření na místě
    attribute authorized { xsd:boolean },

    ## Časové informace vč. časové zóny (buď "Z" nebo (+|-)HH:MM) )
    element MTIME {

        ## Čas vytvoření zprávy
        element TGEN { type-datetime-with-timezone },

        ## Počátek platnosti
        element TSTA { type-datetime-with-timezone },

        ## Konec platnosti
        element TSTO { type-datetime-with-timezone }
    },

    ## Textový popis události vyjadřující KDE, CO, KDY, případně JAK dlouho
    element MTXT {

        ## Popis
        text
    },

    ## Informace o události
    element_MEVT,

    ## Informace o místu události
    element_MLOC,

    ## Informace o místě události v souladu s republikovým číselníkem UIRADR
    element_MDST,

    ## Trasy objížděk
    ##
    ## Povinný pouze v případě, že existuje alespoň jedna objízdná trasa
    element_DIVLOC?
}

element_MEVT =

## Zapouzdřuje veškeré informace o události
element MEVT {

    ## Popis události dle ČSN EN ISO 14819-1 (ALERT-C)
    element TMCE {

        ## Urgentnost (dle ALERT-c) událost zadaná operátorem
        attribute urgencyvalue {

			## normální
			"N" |

			## standardní
			"U" |

			## mimořádně urgentní
			"X"
		},

        ## Ovlivněné směry provozu (z pohledu ALERT-C)
        attribute directionalityvalue {

			## ovlivněn výchozí směr provozu
			"1" |

			## ovlivněny oba směry provozu
			"2"
		},

        ## Dlouhodobost události dle ALERT-C
        attribute timescalevalue {

			## krátkodobá
			"D" |

			## ukončující krátkodobou
			"(D)" |

			##  déletrvající
			"L" |

			## ukončující déletrvající
			"(L)"
		},

        ## Doporučení událost objet
        ##
        ## Pokud je "true", je doporučováno událost objet
        attribute diversion { xsd:boolean },

        ## Popis události pomocí ALERT-C číselníku událostí. Multiplicita: 1-3.
        element_EVI+,

        ## Popis doplňkové události 
        ## včetně např. rychlostního limitu a délky ovlivněného úseku
		##
        ## Kódy viz ALERT-C.
        element SPI {

            ## Číslo doplňkové události dle ALERT-C
            attribute supinfocode { xsd:positiveInteger },

            ## Text doplňkové události dle ALERT-C
            attribute supinfotext { text },

            ## Kód rychlostního limitu dle ALERT-C
            ##
            ## Atribut je povinný, byl-li údaj zadán
            attribute speedlimit { xsd:integer { minInclusive="1" maxInclusive="26" } }?,

            ## Kód délky ovlivněné komunikace podle katalogu ALERT-C
            ##
            ## 0 = úsek je delší než 100 km;
            ## 1-10 = úsek je 1 až 10 km (1 km interval);
            ## 11-15 = úsek je 12 až 20 km (2 km interval);
            ## 16-31 = úsek je 25 až 100 km (5 km interval);
            ##
            ## Atribut je povinný, byl-li údaj zadán
            attribute length { xsd:integer { minInclusive="0" maxInclusive="31" } }?
        }?,

        ## Doporučení místo události objet
		##
		## Povinné, pokud `//TMCE/@diversion` == true
        element DIV {

            ## Kód doplňkové informace k objížďce dle ALERT-C
            attribute diversioncode { xsd:positiveInteger }?,

            ## Text doplňkové informace dle jejího kódu ALERT-C
            ##
            ## Není-li zadán diversioncode je impliciní hodnota:
			## "Doporučuje se objetí místa události"
            attribute diversiontext { text }
        }?,

        ## Text zprávy složený z hodnot jednotlivých událostí 
        ## z ALERT-C katalogu událostí
        element TXTMCE {

            ## Vlastní zpráva
            text
        }
    }?,

    ## Volný text zadaný operátorem
    ##
    ## Element je povinný pouze v případě, kdy je obsažen ve zprávě
    element OTXT {

        ## Vlastní text
        text
    }?,


    ## Suffix textu události – text "Vydal"
    ##
    ## Příklad: "Vydal: Městský úřad Kralupy nad Vltavou"
    element STXT {

        ## Vlastní text
        text
    }?
}

element_MDST =

## Informace o místě události v souladu s republikovým číselníkem UIRADR
##
## Povinný v případě, že zpráva obsahuje české události s vyplněnou 
## informací o místu události v souladu s republikovým číselníkem UIRADR.
element MDST {

    ## Informace o názvu a kódu administrativních jednotek
    ##
    ## Může se opakovat v případě, že událost leží na hranici 
    ## administrativní jednotky, resp. v několika jednotkách
    element DEST {

        ## Název státu
        attribute CountryName { text },

        ## Název městské části
        ##
        ## poviný pouze v případě, že událost leží v obci s tímto členěním
        attribute TownDistrictName { text }?,

        ## Kód městské části
        ##
        ## povinný pouze v případě, že událost leží v obci s tímto členěním
        attribute TownDistrictCode { xsd:integer }?,

        ## Název obce
        attribute TownName { text },

        ## Kód obce
        attribute TownCode { xsd:positiveInteger },

        ## Název okresu
        attribute TownShip { text },

        ## Kód okresu
        attribute TownShipCode { xsd:positiveInteger },

        ## Název kraje
        attribute RegionName { text },

        ## Kód okresu
        attribute RegionCode { xsd:positiveInteger },

        ## Informace o administrativní jednotce ulice
        ##
        ## Povinný, jestliže se jedná o liniovou událost.
        ##
        ## Pro každý úsek, kterým událost prochází, bude generován 
        ## jeden element STRE.
        ##
        ## Opakující se názvy jsou vyloučeny.
        element STRE {

            ## Název ulice
            ##
            ## Povinný, je-li k dispozici v datech podkladové sítě komunikací
            attribute StreetName { text }?,

            ## Kód ulice
            ##
            ## Povinný, je-li k dispozici v datech podkladové sítě komunikací
            attribute StreetCode { xsd:positiveInteger }?
        }*,

        ## Informace o komunikaci vzhledem k UIRADR
        ##
        ## Povinný, jestliže se jedná o liniovou událost.
        ##
        ## Pro každý úsek, kterým událost prochází bude generován jeden node ROAD.
        ##
        ## Opakující se názvy jsou vyloučeny.
        element ROAD {

            ## Číslo komunikace
            ##
            ## Povinný, je-li k dispozici v datech podkladové sítě komunikací
            attribute RoadNumber { text }?,

            ## Třída komunikace
            attribute RoadClass {

                ## dálnice
                "0" |

                ## silnice 1.třídy
                "1" |

                ## silnice 2.třídy
                "2" |

                ## silnice 3.třídy
                "3" |

                ## ostatní silnice
                "4" |

                ## rychlostní komunikace
                "5"
            }
        }*
    }+
}

element_SNTL =

## Místo události popsané odkazy na úseky z datové sady sítě komunikací
element SNTL {

    ## Počet ovlivněných úseků sítě komunikací
    attribute count { xsd:positiveInteger },

    ## Souřadnice počátku události.
    ## U linivé události souřadnice značí místo, které je z pohledu směru 
    ## jízdy nejvíce vpředu, např. místo nehody, za kterou vznikla kolona.
    ##
    ## U ostatních typů lokací může být souřadnice kdekoliv se to zdá
    ## relevantní. 
    element COORD { type-coords-all },

    ## Souřadnice konce události.
    ## U linivé události souřadnice značí místo, které je z pohledu směru jízdy
    ## nejvíce vzadu, např. místo, kde se vozidlo připojí do vznikající kolony.
    ##
    ## U ostatních typů lokací může být souřadnice kdekoliv se to zdá relevantní.
    element COORDEND { type-coords-all },

    ## Seznam identifikátorů ovlivněných úseků
    element STEL {

        ## Identifikátor úseku v síti komunikací
        attribute el_code { xsd:positiveInteger },

        ## Vztah mezi směrem dopravního proudu a orientací geometrie úseku 
        ## sítě komunikací
        attribute el_dir {
			## souběžné
			"+" |

			## protiběžné
			"-"
		},

        ## Pořadí úseku od počátku dopravního problému (např. od nehody ke
        ## konci kolony)
        attribute order { xsd:positiveInteger }
    }+,

    ## Relativní vzdálenost počátku (resp. konce) události na prvním (resp.
    ## posledním) úseku od počátku daného úseku.
    ##
    ## Vzdálenost je vztažena k celkové délce daného úseku, hodnota je tedy v
    ## intervalu 0 až 1.
    ##
    ## Za úsek se považuje geometrie popsaná jedním elementem STEL.
    element STEP {

        ## Relativní vzdálenost počátku události od počátku prvního úseku
        attribute begin {
            xsd:decimal { minInclusive="0" maxInclusive="1" }
        },

        ## Relativní vzdálenost konce události od počátku posledního úseku
        attribute end {
            xsd:decimal { minInclusive="0" maxInclusive="1" }
        }
    }
}

element_OLR =

## Popis místa metodou OpenLR (viz https://www.openlr-association.com/method.html)
element OLR {
    element_XMLLocationReference | element_ERR
}


## ==========================================================

## XMLLocationReference dle TomTom XML schematu
element_XMLLocationReference = element openlr:XMLLocationReference { type-openlr-element }

## Jakýkoliv element z namespace openlr
type-openlr-element = ( element openlr:* { type-openlr-element } | text )*

## Element deklarující, že očekávaný obsah se nepodařilo vytvořit.
element_ERR = element ERR { text }

element_EVI =
    ## Popis (části) události pomocí katalogu událostí ALERT-C
    element EVI {

        ## Kód události dle ALERT-C
        attribute eventcode { xsd:positiveInteger },

        ## Aktualizační třída události
        attribute updateclass { xsd:positiveInteger },

        ## Kvantifikátor události dle ALERT-C
        ##
        ## Povinný, pokud byl použit kvantifikátor
        attribute quantifier { xsd:nonNegativeInteger }?,

        ## Pořadí zprávy
        attribute eventorder { xsd:integer { minInclusive="1" maxInclusive="3" } },

        ## Typ kvantifikátoru události podle číselníku ALERT-C
        ## Rozsah hodnot odpovídá číselníku ALERT-C
        attribute quantifiertype {
            xsd:integer { minInclusive="1" maxInclusive="12" }
        }?,

        ## Název kategorie kódu události dle ALERT-C (ČSN EN ISO 14819-2)
        element TXUCL {

            text
        },

        ## Fráze kódu události dle ALERT-C (ČSN EN ISO 14819-2)
        element TXEVC {

            text
        }
    }

element_TMCL =
    ## Popis místa pomocí ALERT-C
    element TMCL {

        ## Číselný kód primární TMC lokace
        attribute primarycode { xsd:positiveInteger },

        ## Rozsah lokace
        attribute extent { xsd:integer { minInclusive="0" maxInclusive="32" } },

        ## Směr dopravního problému
        attribute direction {

            ## pozitivní TODO - zrevidovat
            "+" |

            ## negativní TODO - zrevidovat
            "-"
        },

        ## Kód komunikace v lokalizační databázi
        attribute roadid { xsd:positiveInteger }
    }

element_CHAIN =
    ## Poloha události vyjádřená názem komunikace a počátečním 
    ## a koncovým provozním staničením.
    ##
    ## Udává se pouze na směrově dělených rychlostních komunikacích. 
    ## Uvede se jednou pro každou relevantní komunikaci.
    element CHAIN {

        ## Název komunikace
        ##
        ## Příklad: "D1"
        attribute road { text },

        ## Hodnota provozního staničení počátku události.
        ##
        ## Pokud je událost orientována ve směru vzrůstajícího provozního
        ## staničení, pak je hodnota "from" menší než hodnota "to".
        attribute from { xsd:decimal },

        ## Hodnota provozního staničení konce události
        attribute to { xsd:decimal },

        ## Orientace dopravního proudu či události vzhledem ke vzrůstajícímu
        ## staničení komunikace
        attribute direction {

            ## dopravní proud či událost je orientován ve směru vzrůstajícího
            ## staničení
            "aligned" |

            ## dopravní proud či událost je orientován proti směru
            ## vzrůstajícího staničení
            "opposite"
        }
    }

type-guid-8-4-4-4-12 =
	## GUID ve formátu 8-4-4-4-12 bez složených závorek.
	xsd:string {
		maxLength="38"
		pattern="[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}"
	}

type-guid-max38 =
	## ID s maximální délkou 38 znaků. Není omezeno jen na hexadecimální znaky.
	xsd:string { maxLength="38" pattern="(\s*[^\s]\s*)+" }

type-nonZeroDecimal = xsd:decimal { minExclusive="0.0" }

type-AlertCEventTableVersion =
	## verze ALERT-C katalogu událostí
	type-nonZeroDecimal

type-AlertCLocationTableVersion =
	## Verze ALERT-C lokační tabulky
	type-nonZeroDecimal

type-country =
	## Kód země dle ISO 3166-1 Alpha-2 (jen země sousedící s ČR)
	"CZ" | "AT" | "DE" | "SK" | "PL"

type-snet-type =
	## Typ mapového podkladu

	## StreetNet
	"SN" |

	## Global Network
	"GN"

type-networkVersion =
	## Verze mapových dat pro silniční síť
	type-nonZeroDecimal

type-geometry_type =
    ## Typ geometrie
    ##
    ## point – bodová geometrie
    "point" |

    ## continuous – spojitá geometrie
    "continuous" |

    ## non-continuous – nespojitá geometrie
    "non-continuous"

type-coord-sjtsk-x = xsd:integer { minInclusive="-1000000" maxInclusive="0"}
type-coord-sjtsk-y = xsd:integer { minInclusive="-1500000" maxInclusive="0"}
type-coord-wgs84-lat = xsd:decimal { minInclusive="48.0" maxInclusive="52.0"}
type-coord-wgs84-lon = xsd:decimal { minInclusive="12.0" maxInclusive="19.0"}

type-coords-all = (
        ## Souřadnice x v systému S-JTSK
        attribute x { type-coord-sjtsk-x },

        ## Souřadnice y v systému S-JTSK
        attribute y { type-coord-sjtsk-y },

        ## Souřadnice lat (zeměpisná šířka) v systému WGS-84
        attribute lat { type-coord-wgs84-lat },

        ## Souřadnice lon (zeměpisná délka) v systému WGS-84
        attribute lon { type-coord-wgs84-lon }
        )
type-datetime-with-timezone =
	## xsd:dateTime vč. časové zóny
	xsd:dateTime { pattern=".*(Z|([+\-]\d{2}:\d{2}))" }
