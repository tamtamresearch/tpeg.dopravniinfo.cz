Příloha B: Reprezentativní XML Schéma RELAX NG¶

K popisu struktury XML Je použito RELAX NG schéma v kompaktním formátu. Doporučujeme zkusit dané schéma přečíst, protože se nám tato forma jeví jako nejkompaktnější a nejčitelnější z dostupných typů schémat.

Pokud však trváte na použití W3C XML Schema 1.0, přikládáme i tento typ schématu, které je vytvořeno konverzí z RELAX NG.

RELAX NG Compact Syntax Schema (RNC)¶

RNC schéma popisuje očekávané struktury XML dokumentu.

Kořenový element je uveden pomocí start =.

Všechny další <SomeName> = elementy definují tzv. vzory.

V našem schématu jsou následující sekce:

  • prolog (deklarace jmenného prostoru)

  • vložení jiných schéma (není využito v tomto případě)

  • definice kořenového elementu (start =)

  • vzory pro elementy

  • vzory pro atributy

  • vzory pro datové typy

Komentáře ## poskytují více detailů k významu jednotlivých částí.

TrafficInformation.rnc - kořen schématu¶

default namespace = ""

start =

## Zapouzdřuje celý dokument pro výměnu informací
element DOC {

    ## Verze XML definice
    attribute version { type-formatVersion },

    ## Jednoznačný identifikátor dokumentu
    ##
    ## GUID může mít různý tvar (6-4, 6-4-4-4-6, 8-4-4-4-12, číslo, ...), vždy je
    ## však unikátní.
    attribute id { text },

    ## Kód země, ve které se události obsažené ve zprávě nachází
    ##
    ## V jedné zprávě se nesmí nacházet událostí z více státu.
    attribute country { type-country },

    ## Použitá datová sada. Datová sada seskupuje údaje související s informací
    ##
    ## Varianty "basic" a "custom" vyřazeny
    attribute DataSet { "extended" },

    ## Informace o subjektech, které si vyměnují XML dokument
    element INF {

        ## Zkratka systému, ve kterém byla dopravní informace pořízena
        ##
        ## Například: "JSDI_NDIC"
        attribute sender { text },

        ## Zkratka systému/subjektu, pro který je zpráva určena napr. HZS
        attribute receiver { text },

        ## Typ přenosu použitý k předání dokumentu
        attribute transmission { "HTTP" | "SMTP" | "FTP" | "SOAP" },

        ## Informace o verzích jednotlivých datových sad
        element DAT {
        
            ## Obsahuje informace o katalogu událostí Alert-C
            element EVTT {

                ## Verze katalogu Alert-C
                attribute version { type-AlertCEventTableVersion },

                ## Jazyková verze
                attribute language { type-language-cz }
            },
            
            ## Obsahuje informace o lokační tabulce
            # TODO: Není obsažen v dokumentaci
            # TODO: Některé vzorky jej neobsahují
            element LOCT {
            
                ## Verze lokační tabulky
                attribute version { type-AlertCLocationTableVersion },
                
                ## Číslo lokační tabulky
                attribute number { xsd:positiveInteger },
                
                ## Číslo země podle číselníku Alert-C
                attribute country { xsd:positiveInteger }
            }?,
            
            ## Informace o datové sadě sítě komunikací, na kterou jsou lokalizovány
            ## dopravní události
            ##
            ## Povinný v případě, že zpráva obsahuje lokalizaci na síť komunikací
            element SNET {
                 
                ## Informace o typu použité sítě komunikací
                ##
                ## "SN" = streetnet "GN" = GlobalNetwork
                attribute type { type-snet-type },

                ## Informace o verzi sítě
                attribute version { type-networkVersion },
                
                ## Země, pro kterou je vytvořena síť komunikací 
                attribute country { "CZ" }
            }?,
            
            ## Informace o datové sadě UIRADR
            # Dokumentace hovoří o elementu UIADR (chyba z přehlédnutí)
            # Dokumentace dovoluje uvézt pouze v případě, že zpráva obsahuje české události s vyplněnou informací o
            # místu události v souladu s republikovým číselníkem UIRADR.
            element UIRADR {
            
                ## Informace o verzi struktury
                attribute structure { text },
                
                ## Informace o verzi dat
                attribute version { type-nonZeroDecimal },
                
                ## Datum vydání verze dat
                attribute date { xsd:date }?
            }
        }
    },
    
    ## Zapouzdřuje veškeré informace o dopravních informacích
    element MJD {
    
        ## Počet dopravních informací, které jsou zapouzdřeny v MJD
        attribute count { xsd:positiveInteger },
        
        ## Představuje jednu zprávu v dokumentu
        element-MSG*
    }
}

element-MSG =

## Představuje jednu zprávu v dokumentu
element MSG {

    ## Jednoznačný identifikátor zprávy
    ##
    ## Formát jednotlivých ID se liší podle poskytovatele události.
    attribute id { text },
    
    ## Číslo verze zprávy
    ##
    ## Nová = 1, aktualizace, vždy +1
    # TODO: Existují zprávy s version = "0"
    attribute version { xsd:nonNegativeInteger },
    
    ## Typ poskytované informace
    ##
    ## "TI" – dopravní informace, "TL" – dopravní intenzita 
    # Typ "WCOND" je popsán jako samostatný formát
    attribute type { "TI" | "TL" },
    
    ## Typ poskytované informace z hlediska jejího charakteru
    ##
    ## Rozlišujeme událost na:
    ## plánovanou z hlediska úředního povolení / skutečný stav
    attribute planned { xsd:boolean },

    ## Časové informace o zprávě
    element MTIME {
    
        ## Popisuje (konstantní) formát datetime.
        attribute format { "YYYY-MM-DDThh:mm:ssTZD" },

        ## Čas kdy byla zpráva vygenerována
        element TGEN { xsd:dateTime },
        
        ## Čas od kdy zpráva platí nebo bude platná
        element TSTA { xsd:dateTime },

        ## Čas do kdy bude zpráva platná
        element TSTO { xsd:dateTime }
    },
    
    ## Kompletní textový popis události vyjadřující KDE, CO, KDY, případně JAK dlouho
    element MTXT {
    
        ## Jazyk, ve kterém je zpráva napsaná 
        attribute language { type-language-cz },
        
        ## Popis
        text
    },
    
    ## Informace o události 
    element-MEVT,
    
    ## Informace o místu události
    element-MLOC,
    
    ## Informace o místě události v souladu s republikovým číselníkem UIRADR
    # Dokumentace povoluje vynechat v případě, že zpráva neobsahuje české události s vyplněnou informací o
    # místu události v souladu s republikovým číselníkem UIRADR
    element-MDST?,
    
    ## Informace o trase objížďky
    ##
    ## Povinný pouze v případě, je-li vyplněna trasa objížďky
    element-DIVLOC?
}

element-MEVT =

## Zapouzdřuje veškeré informace o události 
element MEVT {

    ## Informace o události - atributy jsou vázány na normu 14819-1
    ##
    ## Povinný v případě, že zpráva obsahuje informace v kódování ALERT-C
    element TMCE {
        
        ## Určuje skutečnou hodnotu urgentnosti události zadané operátorem dle ALERT-C
        ## "N" normální "U" je standardní "X" jsou mimořádně urgentní 
        attribute urgencyvalue { "N" | "U" | "X" },
        
        ## Určuje skutečnou hodnotu směrovosti události zadané operátorem vzhledem k ovlivněnému směru provozu dle ALERT-C
        ##
        ## "1": událostí je ovlivněn jeden směr provozu 
        ## "2": událostí jsou ovlivněny oba dva směry
        attribute directionalityvalue { "1" | "2" },
        
        ## Určuje skutečnou hodnotu typu trvání události zadané operátorem dle ALERT-C
        ##
        ## "D" nebo "(D)" krátkodobá "L" nebo "(L)" déletrvající
        # TODO: V dokumentaci uveden format: "kladné celé číslo"
        # TODO: V reálných datech se vyskytuje i prázdná hodnota
        attribute timescalevalue { type-time-scale-value },
        
        ## Textový popis doby platnosti celku zprávy – viz číselník Alert-C
        ##
        ## Povinný v případě je-li čas trvání stanoven výběrem z číselníku 
        attribute durationtext { text }?,
        
        ## Určuje jestli je doporučena objížďka či ne
        ##
        ## "true" – doporučuje objížďku, "false" – nedoporučuje objížďku
        # TODO: Dokumentace popisuje "True" a "False" (velká písmena)
        attribute diversion { xsd:boolean },
        
        ## Informace o čísle události v katalogu událostí o jeho aktualizační třídě,
        ## kvantifikátoru, pořadí (pokud je zpráv více za sebou) včetně textového popisu aktualizační třídy a události.
        ##
        ## Může se max. 3x opakovat
        element EVI {
        
            ## Číslo z katalogu událostí Alert-C
            attribute eventcode { xsd:positiveInteger },

            ## Informace o aktualizační třídě události
            ##
            ## Odpovídá číselníku Alert-C
            attribute updateclass { xsd:positiveInteger },
                
            ## Kvantifikátor události podle číselníku Alert-C
            ##
            ## Povinný, pokud byl použit kvantifikátor
            attribute quantifier { xsd:nonNegativeInteger }?,

            ## Pořadí zprávy
            attribute eventorder { xsd:integer { minInclusive="1" maxInclusive="3" } },
            
            ## Textový popis skupiny události podle ALERT-C
            element TXUCL {
            
                ## Jazyk ve kterém text psaný
                attribute language { type-language-cz },
                
                ## Vlastní popis
                text
            },
            
            ## Textový popis kódu události podle ALERT-C
            element TXEVC {
            
                ## Jazyk, ve kterém text napsaný
                attribute language { type-language-cz },
                
                ## Vlastni popis
                text
            }
        }+,
        
        ## Informace o čísle doplňkové události, rychlostním limitu a délce ovlivněného úseku komunikace
        ##
        ## Element je povinný, pokud byla zadána doplňková událost
        element SPI {
        
            ## Číslo doplňkové události podle katalogu Alert-C
            attribute supinfocode { xsd:positiveInteger },
            
            ## Text doplňkové události podle katalogu Alert-C
            attribute supinfotext { text },
            
            ## Kód rychlostního limitu podle Alert-C
            ##
            ## Atribut je povinný, byl-li údaj zadán
            attribute speedlimit { xsd:integer { minInclusive="1" maxInclusive="26" } }?,
            
            ## Kód délky ovlivněné komunikace podle katalogu Alert-C
            ##
            ## 0 = úsek je delší než 100 km;
            ## 1-10 = úsek je 1 až 10 km (1 km interval);
            ## 11-15 = úsek je 12 až 20 km (2 km interval);
            ## 16-31 = úsek je 25 až 100 km (5 km interval);
            ##
            ## Atribut je povinný, byl-li údaj zadán
            attribute length { xsd:integer { minInclusive="0" maxInclusive="31" } }?
        }?,
        
        ## Informace o objížďce dle ALERT-C.
        ##
        ## Element je povinný, je-li doporučena objížďka 
        element DIV {
        
            ## Kód doplňkové informace dle ALERT-C
            ##
            ## Atribut je povinný, pokud je specifikována doplňková informace 
            attribute diversioncode { xsd:positiveInteger }?,
            
            ## Text doplňkové informace dle jejího kódu ALERT-C
            ##
            ## Není-li zadán diversioncode je hodnota "Doporučuje se objetí místa události"
            attribute diversiontext { text },
            
            ## Jazyk ve kterém je zpráva napsaná 
            attribute language { type-language-cz }
        }?,
        
        ## Text zprávy složený z hodnot jednotlivých událostí z ALERT-C katalogu událostí
        element TXTMCE {
        
            ## Jazyk ve kterém je zpráva napsaná
            attribute language { type-language-cz },

            ## Vlastní zpráva
            text
        }
    }?,
    
    ## Volný text zadaný operátorem
    ##
    ## Element je povinný pouze v případě, kdy je obsažen ve zprávě
    element OTXT {

        ## Vlastní text
        text
    }?
}

element-MLOC =

## Informace o místu události
element MLOC {

    ## Textová informace popisující místo události nebo trasu objížďky
    element-TXPL,
    
    ## Informace o místu události zakódované pomocí lokalizační databáze
    element TMCL {

        ## Číselný kód primární TMC lokace
        ##
        ## Musí být v seznamu kódů lokalizačních bodů 
        # TODO: V dokumentaci uveden jako "Primarycode"
        attribute primarycode { xsd:positiveInteger },
        
        ## Rozsah lokace
        # TODO: V dokumentaci uveden jako "Extent"
        attribute extent { xsd:integer { minInclusive="0" maxInclusive="32" } },
        
        ## Udává směr dopravního problému
        # TODO: V dokumentaci uveden jako "Direction"
        attribute direction { "+" | "-" },
        
        ## Kód komunikace v lokalizační databázi
        # TODO: V dokumentaci uveden jako "Roadid"
        attribute roadid { xsd:positiveInteger }
    }?,
    
    ## Informace o místu události zadané pomocí úseků použité sítě komunikací
    element SNTL {
    
        ## Název použitého souřadného systému
        ##
        ## Požadovaný typ souřadného systému je nutno specifikovat v požadavcích při realizaci odběru informací
        attribute coordsystem { type-coord-system },
        
        ## Počet ovlivněných úseků sítě komunikací
        # TODO: V dokumentaci uveden jako "Count"
        attribute count { xsd:positiveInteger },
        
        ## Souřadnice události nebo zpravodajské oblasti v souřadnicovém systému.
		## U linivé události souřadnice značí počátek události ve smyslu směru dopravy 
		## (místo ke kterému dorazí řidič jako první, v TMC je to sekundární lokace).
        ##
        ## Definovaném v DOC\MJD\MSG\MLOC\SNTL – atribut „coordsystem“
        element COORD {
        
            ## Souřadnice x
            attribute x { xsd:decimal },

            ## Souřadnice y
            attribute y { xsd:decimal }
        },
        
        ## Seznam čísel ovlivněných úseku
        element STEL {
        
            ## Identifikátor úseku v síti komunikací
            attribute el_code { xsd:positiveInteger }
        }+
    }
}

element-MDST =

## Informace o místě události v souladu s republikovým číselníkem UIRADR
##
## Povinný v případě, že zpráva obsahuje české události s vyplněnou informací o
## místu události v souladu s republikovým číselníkem UIRADR.
element MDST {

    ## Informace o názvu a kódu administrativních jednotek
    ##
    ## Může se opakovat v případě, že událost leží na hranici administrativní
    ## jednotky, resp. v několika jednotkách
    element DEST {
    
        ## Název státu
        attribute CountryName { text },
        
        ## Název městské části
        ##
        ## Atribut je poviný pouze v případě, že událost leží v obci s tímto členěním
        attribute TownDistrictName { text }?,
        
        ## Kód městské části
        ## Atribut je povinný pouze v případě, že událost leží v obci s tímto členěním
        attribute TownDistrictCode { xsd:integer }?,

        ## Název obce
        attribute TownName { text },
        
        ## Kód obce
        attribute TownCode { xsd:positiveInteger },
        
        ## Název okresu
        attribute TownShip { text },
        
        ## Kód okresu
        attribute TownShipCode { xsd:positiveInteger },
        
        ## Název kraje
        attribute RegionName { text },
        
        ## Kód okresu
        attribute RegionCode { xsd:positiveInteger },
        
        ## Informace o administrativní jednotce ulice
        ##
        ## Povinný, jestliže se jedná o liniovou událost.
        ##
        ## Pro každý úsek, kterým událost prochází, bude generován jeden node STRE.
        ## Opakující se názvy jsou vyloučeny.
        element STRE {
        
            ## Název ulice
            ##
            ## Povinný, je-li k dispozici v datech podkladové sítě komunikací
            attribute StreetName { text }?,
            
            ## Kód ulice
            ##
            ## Povinný, je-li k dispozici v datech podkladové sítě komunikací
            attribute StreetCode { xsd:positiveInteger }?
        }*,
        
        ## Informace o komunikaci vzhledem k UIRADR
        ##
        ## Povinný, jestliže se jedná o liniovou událost.
        ##
        ## Pro každý úsek, kterým událost prochází bude generován jeden node ROAD.
        ## Opakující se názvy jsou vyloučeny.
        element ROAD {
        
            ## Číslo komunikace
            ##
            ## Povinný, je-li k dispozici v datech podkladové sítě komunikací
            attribute RoadNumber { text }?,
            
            ## Třída komunikace
            attribute RoadClass {

                ## dálnice
                "0" |

                ## silnice 1.třídy
                "1" |

                ## silnice 2.třídy
                "2" |

                ## silnice 3.třídy
                "3" |

                ## ostatní silnice
                "4" |

                ## rychlostní komunikace
                "5"
            }
        }*
    }+
}

element-DIVLOC =

## Informace o trase objížďky
##
## Povinný pouze v případě, je-li vyplněna trasa objížďky
element DIVLOC {

    ## Lokalizace a detailní informace o trase objížďky
    element DIVROUTE {
        
        ## Podrobnější informace týkající se charakteru objížďky
        attribute description { text }
    }+
}

element-TXPL =

## Textová informace popisující místo události nebo trasu objížďky 
element TXPL { text }

type-version = xsd:positiveInteger

type-AlertCLocationTableVersion = type-nonZeroDecimal

type-language-cz = "CZ"

type-country = "CZ" | "AT" | "DE" | "SK" | "PL"

type-snet-type = "SN" | "GN"

type-networkVersion = type-nonZeroDecimal

type-time-scale-value = "D" | "(D)" | "L" | "(L)" | ""

type-coord-system = "S-JTSK" | "WGS-84"
